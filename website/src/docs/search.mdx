## Facets

Blah blah

<InlineGraphiQL
  query={`query ($q: SearchQuery!, $facets: [FacetField!]!) {
      search(q: $q) {
        hitcount
        facets(facets: $facets) {
          name
          values(limit: 10) {
            term
            count
          }
        }
      }
    }`}
  variables={{
    q: {
      all: "harry",
    },
    facets: ["language"],
  }}
/>
